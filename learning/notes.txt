#!/bin/bash
# ================================================
# Improved & Expanded Bash Scripting Notes
# ================================================

# Always start your scripts with this:
#!/bin/bash

# Make script executable: chmod +x script.sh

### ====================== Basic Commands ======================
echo "Hello World"                    # Print text
echo -e "Line1\nLine2"                # Enable escapes (\n, \t, etc.)

read -p "Enter your name: " name      # Get user input
printf "Name: %s, Age: %d\n" "Alice" 25   # Formatted output

sleep 2                               # Pause for 2 seconds
exit 0                                # Exit successfully (1 = error)


### ====================== Variables ======================
name="Tommy"
echo "Hello $name"
echo "Hello ${name}"                  # Safer syntax

# Positional arguments
echo "Script: $0"
echo "First arg: $1"
echo "All args: $@"
echo "Number of args: $#"

# Command substitution
user=$(whoami)
today=$(date +%Y-%m-%d)

# Built-in variables
echo "User: $USER"
echo "Home: $HOME"
echo "Current dir: $PWD"
echo "Shell: $SHELL"
echo "Last command status: $?"
echo "Random number: $RANDOM"


### ====================== Arithmetic ======================
result=$(( 5 + 3 * 2 ))     # 11
(( counter++ ))             # Increment
echo $(( 2**4 ))            # Power = 16


### ====================== Conditionals ======================
if [[ $x -gt 0 ]]; then
    echo "Positive"
elif [[ $x -lt 0 ]]; then
    echo "Negative"
else
    echo "Zero"
fi

# File tests
if [[ -f "file.txt" ]]; then
    echo "File exists"
elif [[ -d "mydir" ]]; then
    echo "Directory exists"
fi

# String tests
if [[ -z "$var" ]]; then
    echo "Variable is empty"
fi


### ====================== Case Statement ======================
case $option in
    start|Start) echo "Starting...";;
    stop)        echo "Stopping...";;
    restart)     echo "Restarting...";;
    *)           echo "Unknown option";;
esac


### ====================== Loops ======================
# For loop - range
for i in {1..5}; do
    echo "Iteration $i"
done

# For loop - list
for name in Alice Bob Charlie; do
    echo "Hello $name"
done

# While loop
count=1
while [[ $count -le 5 ]]; do
    echo "Count: $count"
    ((count++))
done

# Until loop
until [[ $count -gt 5 ]]; do
    echo "Count: $count"
    ((count++))
done

# C-style for loop
for ((i=0; i<5; i++)); do
    echo $i
done


### ====================== Functions ======================
greet() {
    local name=$1          # Local variable
    echo "Hello $name!"
}

greet "World"


### ====================== Best Practices ======================
# At the top of your script:
set -e          # Exit on error
set -u          # Treat unset variables as error
set -o pipefail # Catch errors in pipes

# Always quote variables
echo "Value: $var"          # Good: "$var"

# Check arguments
[[ $# -eq 1 ]] || { echo "Usage: $0 <name>"; exit 1; }


### ====================== Useful Commands ======================
# Text processing: grep, awk, sed, cut, sort, uniq, wc
# Files: ls, mkdir, cp, mv, rm, touch, find
# Network: curl, wget
# Others: date, cal, head, tail, cat

EOF
